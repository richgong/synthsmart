76 void IGraphicsAGG::Rasterizer::RasterizePattern(agg::trans_affine transform, const IPattern& pattern, const IBlend* pBlend, EFillRule rule)
77 {
78  mRasterizer.filling_rule(rule == kFillWinding ? agg::fill_non_zero : agg::fill_even_odd );
79 
80  switch (pattern.mType)
81  {
82  case kSolidPattern:
83  {
84  RendererSolid renderer(mRenBase);
85 
86  const IColor &color = pattern.GetStop(0).mColor;
87  renderer.color(AGGColor(color, pBlend));
88 
89  // Rasterize
90 
91  Rasterize(renderer, AGGBlendMode(pBlend));
92  }
93  break;
94 
95  case kLinearPattern:
96  case kRadialPattern:
97  {
98  // Common gradient objects
99 
100  const float* xform = pattern.mTransform;
101 
102  agg::trans_affine gradientMTX(xform[0], xform[1] , xform[2], xform[3], xform[4], xform[5]);
103  ColorArrayType colorArray;
104 
105  // Scaling
106 
107  gradientMTX = transform * gradientMTX * agg::trans_affine_scaling(512.0);
108 
109  // Make gradient lut
110 
111  colorArray.remove_all();
112 
113  for (int i = 0; i < pattern.NStops(); i++)
114  {
115  const IColorStop& stop = pattern.GetStop(i);
116  float offset = stop.mOffset;
117  colorArray.add_color(offset, AGGColor(stop.mColor, pBlend));
118  }
119 
120  colorArray.build_lut();
121 
122  // Rasterize
123 
124  if (pattern.mType == kLinearPattern)
125  {
126  GradientRasterizeAdapt(*this, pattern.mExtend, agg::gradient_x(), gradientMTX, colorArray, AGGBlendMode(pBlend));
127  }
128  else
129  {
130  GradientRasterizeAdapt(*this, pattern.mExtend, agg::gradient_radial_d(), gradientMTX, colorArray, AGGBlendMode(pBlend));
131  }
132  }
133  break;
134  }
135 }

182 bool checkTransform(const agg::trans_affine& mtx)
183 {
184  double mtx_copy[6];
185  const double epsilon = agg::affine_epsilon;
186  mtx.store_to(mtx_copy);
187 
188  if (!agg::is_equal_eps(mtx_copy[4] - floor(mtx_copy[4]), 0.0, epsilon))
189  return false;
190  if (!agg::is_equal_eps(mtx_copy[5] - floor(mtx_copy[5]), 0.0, epsilon))
191  return false;
192 
193  agg::trans_affine mtx_without_translate;
194 
195  mtx_copy[4] = 0.0;
196  mtx_copy[5] = 0.0;
197  mtx_without_translate.load_from(mtx_copy);
198 
199  return mtx_without_translate.is_identity();
200 }
