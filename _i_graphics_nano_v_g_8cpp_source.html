68 NVGpaint NanoVGPaint(NVGcontext* context, const IPattern& pattern, const IBlend* pBlend)
69 {
70  NVGcolor icol = NanoVGColor(pattern.GetStop(0).mColor, pBlend);
71  NVGcolor ocol = NanoVGColor(pattern.GetStop(pattern.NStops() - 1).mColor, pBlend);
72 
73  // Invert transform
74 
75  float inverse[6];
76  nvgTransformInverse(inverse, pattern.mTransform);
77  float s[2];
78 
79  nvgTransformPoint(&s[0], &s[1], inverse, 0, 0);
80 
81  if (pattern.mType == kRadialPattern)
82  {
83  return nvgRadialGradient(context, s[0], s[1], 0.0, inverse[0], icol, ocol);
84  }
85  else
86  {
87  float e[2];
88  nvgTransformPoint(&e[0], &e[1], inverse, 1, 0);
89 
90  return nvgLinearGradient(context, s[0], s[1], e[0], e[1], icol, ocol);
91  }
