291  IPattern GetSVGPattern(const NSVGpaint& paint, float opacity)
292  {
293  int alpha = std::min(255, std::max(int(roundf(opacity * 255.f)), 0));
294 
295  switch (paint.type)
296  {
297  case NSVG_PAINT_COLOR:
298  return IColor(alpha, (paint.color >> 0) & 0xFF, (paint.color >> 8) & 0xFF, (paint.color >> 16) & 0xFF);
299 
300  case NSVG_PAINT_LINEAR_GRADIENT:
301  case NSVG_PAINT_RADIAL_GRADIENT:
302  {
303  NSVGgradient* pGrad = paint.gradient;
304 
305  IPattern pattern(paint.type == NSVG_PAINT_LINEAR_GRADIENT ? kLinearPattern : kRadialPattern);
306 
307  // Set Extend Rule
308  switch (pGrad->spread)
309  {
310  case NSVG_SPREAD_PAD: pattern.mExtend = kExtendPad; break;
311  case NSVG_SPREAD_REFLECT: pattern.mExtend = kExtendReflect; break;
312  case NSVG_SPREAD_REPEAT: pattern.mExtend = kExtendRepeat; break;
313  }
314 
315  // Copy Stops
316  for (int i = 0; i < pGrad->nstops; i++)
317  {
318  int color = pGrad->stops[i].color;
319  pattern.AddStop(IColor(255, (color >> 0) & 0xFF, (color >> 8) & 0xFF, (color >> 16) & 0xFF), pGrad->stops[i].offset);
320  }
321 
322  // Copy transform
323  pattern.SetTransform(pGrad->xform[0], pGrad->xform[1], pGrad->xform[2], pGrad->xform[3], pGrad->xform[4], pGrad->xform[5]);
324 
325  return pattern;
326  }
327  default:
328  return IColor(alpha, 0, 0, 0);
329  }
330  }

332  void RenderNanoSVG(NSVGimage* pImage)
333  {
334  assert(pImage != nullptr);
335 
336  for (NSVGshape* pShape = pImage->shapes; pShape; pShape = pShape->next)
337  {
338  if (!(pShape->flags & NSVG_FLAGS_VISIBLE))
339  continue;
340 
341  for (NSVGpath* pPath = pShape->paths; pPath; pPath = pPath->next)
342  {
343  PathMoveTo(pPath->pts[0], pPath->pts[1]);
344 
345  for (int i = 0; i < pPath->npts - 1; i += 3)
346  {
347  float *p = pPath->pts + i * 2 + 2;
348  PathCurveTo(p[0], p[1], p[2], p[3], p[4], p[5]);
349  }
350 
351  if (pPath->closed)
352  PathClose();
353  }
354 
355  // Fill
356  if (pShape->fill.type != NSVG_PAINT_NONE)
357  {
358  IFillOptions options;
359 
360  if (pShape->fillRule == NSVG_FILLRULE_EVENODD)
361  options.mFillRule = kFillEvenOdd;
362  else
363  options.mFillRule = kFillWinding;
364 
365  options.mPreserve = pShape->stroke.type != NSVG_PAINT_NONE;
366  PathFill(GetSVGPattern(pShape->fill, pShape->opacity), options, nullptr);
367  }
368 
369  // Stroke
370  if (pShape->stroke.type != NSVG_PAINT_NONE)
371  {
372  IStrokeOptions options;
373 
374  options.mMiterLimit = pShape->miterLimit;
375 
376  switch (pShape->strokeLineCap)
377  {
378  case NSVG_CAP_BUTT: options.mCapOption = kCapButt; break;
379  case NSVG_CAP_ROUND: options.mCapOption = kCapRound; break;
380  case NSVG_CAP_SQUARE: options.mCapOption = kCapSquare; break;
381  }
382 
383  switch (pShape->strokeLineJoin)
384  {
385  case NSVG_JOIN_MITER: options.mJoinOption = kJoinMiter; break;
386  case NSVG_JOIN_ROUND: options.mJoinOption = kJoinRound; break;
387  case NSVG_JOIN_BEVEL: options.mJoinOption = kJoinBevel; break;
388  }
389 
390  options.mDash.SetDash(pShape->strokeDashArray, pShape->strokeDashOffset, pShape->strokeDashCount);
391 
392  PathStroke(GetSVGPattern(pShape->stroke, pShape->opacity), pShape->strokeWidth, options, nullptr);
393  }
394  }
